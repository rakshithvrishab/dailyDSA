//Moorâ€™s Voting Algorithm 
class Solution {
    public int majorityElement(int[] A) {
        int majority = A[0];
        int cnt = 1;
        for(int i=1;i<A.length;i++) {
            if(cnt==0){
                majority = A[i];
            }
            if(A[i]==majority){
                ++cnt;
            }
            else{
                --cnt;
            }
        }
        return majority;
    }
}

//TC:O(N) 
//SC:O(1)
